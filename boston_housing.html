def performance_metric2(y_true, y_predict):
    """"""
    ss_res = sum([(y_true[x] - y_predict[x])**2 for x in range(len(y_true))])
    y_true_bar = sum(y_true)/len(y_true)
    ss_tot = sum([(y_true[x] - y_true_bar)**2 for x in range(len(y_true))])
    score = 1 - ss_res/ss_tot

    return score

def fit_model(X, y):
    """  [X,y]"""
    
    cross_validator = KFold(random_state=1)
    
    regressor = DecisionTreeRegressor()

    params = {'max_depth':[i for i in range(1,11)]}

    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(regressor, param_grid=params, scoring=scoring_fnc, cv=cross_validator)

    #  [X,y]
    grid = grid.fit(X, y)

    # 
    return grid.best_estimator_

import copy
def cross_val_score(estimator, X, y, scoring = performance_metric, cv=3):
    """  """
    l = len(X)
    data_x = np.array(X)
    fold_x = data_x.tolist()
    fold_y = y.tolist()
    scores = []
    for i in range(cv):

        test_x = fold_x[i*l/cv:(i+1)*l/cv]
        test_y = fold_y[i*l/cv:(i+1)*l/cv]
        train_x = copy.deepcopy(fold_x)
        train_y = copy.deepcopy(fold_y)
        del(train_x[i*l/cv:(i+1)*l/cv])
        del(train_y[i*l/cv:(i+1)*l/cv])
        estimator.fit(train_x, train_y)
        pred = estimator.predict(test_x)
        scores.append(scoring(test_y, pred))
    return scores

def fit_model2(X, y):
    """  [X,y]"""
    max_depth = [i for i in range(1,11)]
    cross_validator = 3
    score = []
    for i in range(len(max_depth)):
        regressor = DecisionTreeRegressor(max_depth=max_depth[i])
        score.append(sum(cross_val_score(regressor, X, y, scoring = performance_metric, cv=cross_validator))/cross_validator) 
    #
    best_estimator = DecisionTreeRegressor(max_depth= score.index(max(score)) + 1)
    
    return best_estimator
